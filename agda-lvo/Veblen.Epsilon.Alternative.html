ε的另一种表示

我们一般化 ε 的下标.

private variable
  α : Ord

首先, 观察 ω ^^[ suc (ε α) ] ⌜ n ⌝ 的前几层有

{ε_α}^+ = {ε_α}^+

_ : ω ^^[ suc (ε α) ] ⌜ 0 ⌝ ≡ suc (ε α)
_ = refl

ω^{{ε_α}^+} = ε_α × ω

ω^^[sε]1 : ω ^^[ suc (ε α) ] ⌜ 1 ⌝ ≈ ε α * ω
ω^^[sε]1 {α} =                begin-equality
  ω ^ ε α * ω                 ≈⟨ *-congʳ {ω} (ε-fp α) ⟩
  ε α * ω                     ∎

ω^{ω^{{ε_α}^+}} = {ε_α}^ω

ω^^[sε]2 : ω ^^[ suc (ε α) ] ⌜ 2 ⌝ ≈ ε α ^ ω
ω^^[sε]2 {α} =                begin-equality
  ω ^ ω ^^[ suc (ε α) ] ⌜ 1 ⌝ ≈⟨ ^-congˡ ω≥1 ω^^[sε]1 ⟩
  ω ^ (ε α * ω)               ≈˘⟨ ^-*-assoc _ _ _ ⟩
  ω ^ ε α ^ ω                 ≈⟨ ^-congʳ {ω} (ε-fp α) ⟩
  ε α ^ ω                     ∎

ω^{ω^{ω^{{ε_α}^+}}} = {ε_α}^{{ε_α}^ω}

ω^{ω^{ω^{.^{.^{{ε_α}^+}}}}} = {ε_α}^{{ε_α}^{.^{.^ω}}}
