向量的枚举性

combine : 𝕃 A → (n : ℕ) → 𝕃 (𝕍 A n)
combine xs zero = [ [] ]
combine xs (suc n) = map (λ (x , x⃗) → x ∷ x⃗) (xs [×] combine xs n)

项的枚举

termLists : 𝕃ₙ Term
termLists zero = []
termLists (suc n) = termLists n ++ # n ∷ concat (map apps (eᶠ n)) where
  apps : 𝓕 → 𝕃 Term
  apps f = map (f $̇_) (combine (termLists n) ∣ f ∣ᶠ)
