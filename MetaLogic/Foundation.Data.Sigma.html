Σ≡p : isPred P → {u v : Σ A P}
       → (p : u .fst ≡ v .fst) → u ≡ v
Σ≡p pP H = Eq←🧊 $ Σ≡Prop (isPred→🧊 pP) (Eq→🧊 H)

×≡ : {x y : A × B} → fst x ≡ fst y → snd x ≡ snd y → x ≡ y
×≡ refl refl = refl

discrete× : discrete A → discrete B → discrete (A × B)
discrete× dA dB = discreteΣ dA dB
