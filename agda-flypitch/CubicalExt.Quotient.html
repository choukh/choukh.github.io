{-# OPTIONS --cubical --safe #-}

module CubicalExt.Quotient where

open import Cubical.Core.Everything
open import Cubical.Foundations.Prelude using (isSet)

data _/_ {a r} (A : Type a) (R : A → A → Type r) : Type (ℓ-max a r) where
  ⟦_⟧ : A → A / R
  quotient : ∀ a b → R a b → ⟦ a ⟧ ≡ ⟦ b ⟧
  truncate : isSet (A / R)
