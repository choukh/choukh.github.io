bound⊨ : ∀ {Γ φ} → unbound ⟦ Γ ⟧ Free.⊨ unbound φ → Γ ⊨ φ
bound⊨ {Γ} {φ} ⊨ 𝒮 = flip $ λ vld →
  rec (isProp-⊨ˢ _ _) λ x → let 𝓋 = λ _ → x in
    from (realize-iff 𝒮 𝓋 φ) $ ⊨ 𝒮 𝓋 λ φ' →
      rec (Free.isPropRealize _ _ _) $ λ { (ψ , ψ∈Γ , reflId) →
        to (realize-iff 𝒮 𝓋 ψ) (vld ψ ψ∈Γ) }
