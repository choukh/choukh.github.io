理论

Theory : Type (ℓ-suc u)
Theory = 𝒫 Sentence u

isSetTheory : isSet Theory
isSetTheory = isSet𝒫

open 𝒫.SetBased {X = Sentence} isSetFormula
open 𝒫.SetBased2 {X = Sentence} {Y = Free.Formula} isSetFormula Free.isSetFormula
open 𝒫.SetBased {X = Free.Formula} Free.isSetFormula using () renaming (_⨭_ to _Free⨭_)

我们使用以下隐式参数.

private variable
  Γ Δ : Theory
  φ φ₁ φ₂ φ₃ : Sentence

证明

infix 4 _⊢_
_⊢_ : Theory → Sentence → Type (ℓ-suc u)
Γ ⊢ φ = unbound ⟦ Γ ⟧ Free.⊢ unbound φ

我们用更短的 ⊦ 表示 ⊢ 的命题截断.

infix 4 _⊦_
_⊦_ : Theory → Sentence → Type (ℓ-suc u)
Γ ⊦ φ = ∥ Γ ⊢ φ ∥₁

open import CubicalExt.Axiom.ExcludedMiddle using (isPropImplicit)
instance
  isProp-⊦ : isPropImplicit (Γ ⊦ φ)
  isProp-⊦ = squash₁ _ _

虽然我们最终只关心 ⊦, 例如一阶逻辑的完备性将使用 ⊦ 来表达, 但在此之前需要先证明一系列关于 ⊢ 的引理.

weakening : Γ ⊆ Δ → Γ ⊢ φ → Δ ⊢ φ
weakening Γ⊆Δ Γ⊢φ = Free.weakening (⟦⟧⊆⟦⟧ Γ⊆Δ) Γ⊢φ

weakening1 : Γ ⊢ φ₂ → Γ ⨭ φ₁ ⊢ φ₂
weakening1 = weakening ⊆⨭

weakening2 : Γ ⨭ φ₁ ⊢ φ₂ → Γ ⨭ φ₃ ⨭ φ₁ ⊢ φ₂
weakening2 = weakening (⨭⊆⨭ ⊆⨭)

axiom : φ ∈ Γ → Γ ⊢ φ
axiom φ∈Γ = Free.axiom ∣ _ , φ∈Γ , reflId ∣₁

axiom1 : Γ ⨭ φ ⊢ φ
axiom1 = axiom $ inr reflId

axiom2 : Γ ⨭ φ₁ ⨭ φ₂ ⊢ φ₁
axiom2 = axiom $ inl $ inr reflId

导出规则

bound⊢ : Γ ⨭ φ₂ ⊢ φ₁ → unbound ⟦ Γ ⟧ Free⨭ unbound φ₂ Free.⊢ unbound φ₁
bound⊢ = Free.weakening ⟦⨭⟧⊆

⇒ 的补充规则

⇒-intro 在有些书中称为演绎定理 (deduction theorem). 我们这里直接指定为规则. 以下是它的逆命题. 两者结合表明了 Γ ⨭ φ₁ ⊢ φ₂ 与 Γ ⊢ φ₁ ⇒ φ₂ 的等价性.

⇒-elim-to-axiom : Γ ⊢ φ₁ ⇒ φ₂ → Γ ⨭ φ₁ ⊢ φ₂
⇒-elim-to-axiom Γ⊢⇒ = ⇒-elim (weakening1 Γ⊢⇒) axiom1

以下可以认为是 ⇒-elim 的逆命题, 但要注意 → 的两边都要对理论做全称量化. 此外, 满足 ∀ Γ → Γ ⊢ φ 的 φ 又称为恒真式 (tautology). 所以以下命题又称为恒真式的引入规则.

⇒-intro-tauto : (∀ {Γ} → Γ ⊢ φ₁ → Γ ⊢ φ₂) → ∀ {Δ} → Δ ⊢ φ₁ ⇒ φ₂
⇒-intro-tauto {φ₁} ⊢ = ⇒-intro $ bound⊢ $ weakening {Γ = ｛ φ₁ ｝} inr $ ⊢ $ axiom reflId

爆炸律

exfalso : Γ ⊢ ⊥ → Γ ⊢ φ
exfalso = Free.exfalso

tauto-exfalso : Γ ⊢ ⊥ ⇒ φ
tauto-exfalso = Free.tauto-exfalso

∧ 的引入引出规则

∧-intro : Γ ⊢ φ₁ → Γ ⊢ φ₂ → Γ ⊢ φ₁ ∧ φ₂
∧-intro = Free.∧-intro

∧-elimₗ : Γ ⊢ φ₁ ∧ φ₂ → Γ ⊢ φ₁
∧-elimₗ = Free.∧-elimₗ

∧-elimᵣ : Γ ⊢ φ₁ ∧ φ₂ → Γ ⊢ φ₂
∧-elimᵣ = Free.∧-elimᵣ

∨ 的引入引出规则

∨-introₗ : Γ ⊢ φ₁ → Γ ⊢ φ₁ ∨ φ₂
∨-introₗ = Free.∨-introₗ

∨-introᵣ : Γ ⊢ φ₂ → Γ ⊢ φ₁ ∨ φ₂
∨-introᵣ = Free.∨-introᵣ

∨-elim : Γ ⊢ φ₁ ∨ φ₂ → Γ ⨭ φ₁ ⊢ φ₃ → Γ ⨭ φ₂ ⊢ φ₃ → Γ ⊢ φ₃
∨-elim Γ⊢∨ ⊢₁ ⊢₂ = Free.∨-elim Γ⊢∨ (bound⊢ ⊢₁) (bound⊢ ⊢₂)

∨-comm : Γ ⊢ φ₁ ∨ φ₂ → Γ ⊢ φ₂ ∨ φ₁
∨-comm = Free.∨-comm

排中律

LEM : Γ ⊢ φ ∨ ~ φ
LEM = Free.LEM

DNE : Γ ⊢ ~ ~ φ ⇒ φ
DNE = Free.DNE

矛盾律

no-contra : Γ ⊢ φ ∧ ~ φ → Γ ⊢ ⊥
no-contra = Free.no-contra

tauto-no-contra : Γ ⊢ ~ (φ ∧ ~ φ)
tauto-no-contra = Free.tauto-no-contra

⇔ 的引入引出规则

⇔-intro : Γ ⨭ φ₁ ⊢ φ₂ → Γ ⨭ φ₂ ⊢ φ₁ → Γ ⊢ φ₁ ⇔ φ₂
⇔-intro ⊢₁ ⊢₂ = Free.⇔-intro (bound⊢ ⊢₁) (bound⊢ ⊢₂)

⇔-elimₗ : Γ ⊢ φ₁ ⇔ φ₂ → Γ ⨭ φ₁ ⊢ φ₂
⇔-elimₗ ⊢⇔ = ⇒-elim-to-axiom (∧-elimₗ ⊢⇔)

⇔-elimᵣ : Γ ⊢ φ₁ ⇔ φ₂ → Γ ⨭ φ₂ ⊢ φ₁
⇔-elimᵣ ⊢⇔ = ⇒-elim-to-axiom (∧-elimᵣ ⊢⇔)

⇔ 的自反性、对称性和传递性

⇔-refl : Γ ⊢ φ ⇔ φ
⇔-refl = Free.⇔-refl

⇔-sym : Γ ⊢ φ₁ ⇔ φ₂ → Γ ⊢ φ₂ ⇔ φ₁
⇔-sym = Free.⇔-sym

⇒-trans : Γ ⊢ φ₁ ⇒ φ₂ → Γ ⊢ φ₂ ⇒ φ₃ → Γ ⊢ φ₁ ⇒ φ₃
⇒-trans = Free.⇒-trans

⇔-trans : Γ ⊢ φ₁ ⇔ φ₂ → Γ ⊢ φ₂ ⇔ φ₃ → Γ ⊢ φ₁ ⇔ φ₃
⇔-trans = Free.⇔-trans
