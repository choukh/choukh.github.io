data Functions ℒ : ℕ → Set u where
  include  : ∀ {n} → ℒ .functions n → Functions ℒ n
  witness : Formula ℒ 1 → Functions ℒ 0

langStep : Language → Language
langStep ℒ = record { functions = Functions ℒ ; relations = ℒ .relations }

langMorph : ℒ ⟶ langStep ℒ
langMorph = record { funcMorph = Functions.include ; relMorph = id }

langChainObj : Language → ℕ → Language
langChainObj ℒ zero    = ℒ
langChainObj ℒ (suc n) = langStep (langChainObj ℒ n)

langChainMorph : ∀ {x y} → .(x ≤₃ y) → langChainObj ℒ x ⟶ langChainObj ℒ y
langChainMorph {ℒ} {x} {y} x≤y with <-cmp x y
... | tri< (s≤s x≤y-1) _ _ = langMorph ∘ langChainMorph (≤⇒≤₃ x≤y-1)
... | tri≈ _ refl _ = idᴸ

langChainFunctorial : ∀ {x y z : ℕ} .{f₁ : x ≤₃ y} .{f₂ : y ≤₃ z} .{f₃ : x ≤₃ z}
  → langChainMorph {ℒ} f₃ ≡ (langChainMorph f₂ ∘ langChainMorph f₁)
langChainFunctorial {ℒ} {x} {y} {z} {x≤y} {y≤z} {x≤z} with <-cmp x y | <-cmp y z | <-cmp x z
... | tri< _ _ x≰y  | tri< x≤y _ _  | tri≈ _ refl _ = ⊥-elim $ x≰y x≤y
... | tri< sx≤x _ _ | tri≈ _ refl _ | tri≈ _ refl _ = ⊥-elim $ 1+n≰n sx≤x
... | tri≈ _ refl _ | tri< sx≤x _ _ | tri≈ _ refl _ = ⊥-elim $ 1+n≰n sx≤x
... | tri≈ _ refl _ | tri≈ _ refl _ | tri< sx≤x _ _ = ⊥-elim $ 1+n≰n sx≤x
... | tri< (s≤s _) _ _ | tri≈ _ refl _    | tri< (s≤s _) _ _ = refl
... | tri≈ _ refl _    | tri< (s≤s _) _ _ | tri< (s≤s _) _ _ = refl
... | tri< (s≤s x≤y-1) x≢x y≮x | tri< (s≤s y≤z-1) _ _ | tri< (s≤s x≤z-1) _ _ =
  cong (langMorph ∘_) $ trans
    (langChainFunctorial {f₁ = x≤₃y} {f₂ = ≤⇒≤₃ y≤z-1} {f₃ = ≤⇒≤₃ x≤z-1})
    (cong (langChainMorph (≤⇒≤₃ y≤z-1) ∘_) eq) where
      x≤₃y : x ≤₃ y
      x≤₃y with <-cmp x y
      ... | tri< _ _ _ = tt
      ... | tri≈ _ _ _ = tt
      ... | tri> _ _ y<x = y≮x y<x
      eq : langChainMorph x≤₃y ≡ langMorph ∘ langChainMorph (≤⇒≤₃ x≤y-1)
      eq with <-cmp x y
      ... | tri< (s≤s _) _ _ = refl
      ... | tri≈ _ refl _ = ⊥-elim $ x≢x refl
      ... | tri> _ _ y<x  = ⊥-elim $ y≮x y<x
... | tri≈ _ refl _    | tri≈ _ refl _ | tri≈ _ x≡x _ with ℕ-K x≡x refl
... | refl = refl

ℕᴰ : DirectedType
ℕᴰ = record
  { Carrier = ℕ
  ; _~_ = _≤₃_
  ; ~-refl = ≤⇒≤₃ ≤-refl
  ; ~-trans = λ p q → ≤⇒≤₃ $ ≤-trans (≤₃⇒≤ p) (≤₃⇒≤ q)
  ; directed = λ x y → x + y , ≤⇒≤₃ (m≤m+n _ _) , ≤⇒≤₃ (m≤n+m _ _)
  }

langChain : Language → DirectedDiagramLanguage ℕᴰ
langChain ℒ = record
  { obj         = langChainObj ℒ
  ; morph       = langChainMorph
  ; functorial  = langChainFunctorial
  }

witnessOf : Formula ℒ 1 → Constant $ langStep ℒ
witnessOf = Functions.witness

[_witnessing_] : Constant ℒ → Formula ℒ 1 → Sentence ℒ
[_witnessing_] {ℒ} c φ = (∃' φ) ⇒ φ [ const c / 0 ] where
  open FOL.Bounded.Base ℒ
  open FOL.Bounded.Substitution ℒ
