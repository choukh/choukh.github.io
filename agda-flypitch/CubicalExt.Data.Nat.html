{-# OPTIONS --cubical --safe #-}

module CubicalExt.Data.Nat where

open import Cubical.Data.Equality using (eqToPath; pathToEq; pathToEq-eqToPath)
open import Cubical.Data.Nat using (ℕ; isSetℕ)

open import Function using (_$_)
open import Relation.Binary.PropositionalEquality as Eq using (_≡_; cong; sym; trans)
open Eq.≡-Reasoning

ℕ-K : ∀ {n : ℕ} (p : n ≡ n) (q : n ≡ n) → p ≡ q
ℕ-K p q =               begin
  p                     ≡˘⟨ pathToEq $ pathToEq-eqToPath p ⟩
  pathToEq (eqToPath p) ≡⟨ cong pathToEq (pathToEq $ isSetℕ _ _ (eqToPath p) (eqToPath q)) ⟩
  pathToEq (eqToPath q) ≡⟨ pathToEq $ pathToEq-eqToPath q ⟩
  q                     ∎
