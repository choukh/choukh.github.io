序数的定义

data Ord : Set where
  zero : Ord
  suc  : Ord → Ord
  lim  : (ℕ → Ord) → Ord

variable
  α β γ δ : Ord
  m n : ℕ

finord : ℕ → Ord
finord zero = zero
finord (suc n) = suc (finord n)

ω = lim finord

open import Agda.Builtin.FromNat

instance
  _ = Number Ord ∋ record { Constraint = λ _ → ⊤ ; fromNat = λ n → finord n }
  _ = Number ℕ   ∋ record { Constraint = λ _ → ⊤ ; fromNat = λ n → n }

_ = Ord ∋ 233

快速增长层级

variable A : Set

_∘ⁿ_ : (A → A) → ℕ → (A → A)
(F ∘ⁿ zero)  a = a
(F ∘ⁿ suc n) a = (F ∘ (F ∘ⁿ n)) a

module FGH where
  f : Ord → ℕ → ℕ
  f zero n = suc n
  f (suc α) n = (f α ∘ⁿ n) n
  f (lim g) n = f (g n) n

f_0(n) = n + 1 

  f-0-2 : f 0 2 ≡ 3
  f-0-2 = refl

f_1(n) = 2n 

  f-1-2 : f 1 2 ≡ 4
  f-1-2 = refl

f_2(n) = 2^n~n 

  f-2-2 : f 2 2 ≡ 8
  f-2-2 = refl

  f-3-2 : f 3 2 ≡ 2048
  f-3-2 = refl

  f-suc : f (suc α) n ≡ (f α ∘ⁿ n) n
  f-suc = refl

  f-ω : f ω n ≡ f (finord n) n
  f-ω = refl

  f-ω⁺ : f (suc ω) n ≡ (f ω ∘ⁿ n) n
  f-ω⁺ = refl

  f-suc-2 : f (suc α) 2 ≡ f α (f α 2)
  f-suc-2 = refl

无穷迭代

elim : {P : Ord → Set}
  → P zero
  → (∀ α → P α → P (suc α))
  → (∀ f → (∀ n → P (f n)) → P (lim f))
  → ∀ α → P α
elim z s l zero = z
elim z s l (suc α) = s α (elim z s l α)
elim z s l (lim f) = l f λ n → elim z s l (f n)

rec : A → (A → A) → ((ℕ → A) → A) → Ord → A
rec z s l = elim z (λ _ → s) (λ _ → l)

_∘^_ : (Ord → Ord) → Ord → Ord → Ord
(F ∘^ α) β = rec β F lim α

variable
  F : Ord → Ord
  f g h : ℕ → Ord

∘^-zero : F ∘^ zero ≡ id
∘^-zero = refl

∘^-suc : F ∘^ suc α ≡ F ∘ (F ∘^ α)
∘^-suc = refl

∘^-lim : F ∘^ lim f ≡ λ β → lim λ n → (F ∘^ (f n)) β
∘^-lim = refl

iterω : (Ord → Ord) → Ord → Ord
iterω F α = (F ∘^ ω) α

序数算术

infixl 6 _+_
_+_ : Ord → Ord → Ord
α + β = (suc ∘^ β) α

infixl 7 _*_
_*_ : Ord → Ord → Ord
α * β = ((_+ α) ∘^ β) 0

infix 8 _^_
_^_ : Ord → Ord → Ord
α ^ β = ((_* α) ∘^ β) 1

_+ω^_ : Ord → Ord → Ord
α +ω^ zero = suc α
α +ω^ suc β = iterω (_+ω^ β) α
α +ω^ lim f = lim λ n → α +ω^ f n

跳出运算

复合了后继的迭代.

jump : (Ord → Ord) → Ord → Ord
jump F α = ((F ∘ suc) ∘^ α) (F zero)

jump-0 : jump F 0 ≡ F 0
jump-0 = refl

jump-suc : jump F (suc α) ≡ F (suc (jump F α))
jump-suc {F} {α} = begin
  jump F (suc α)                        ≡⟨⟩
  ((F ∘ suc) ∘^ (suc α)) (F zero)       ≡⟨⟩
  (F ∘ suc) (((F ∘ suc) ∘^ α) (F zero)) ≡⟨⟩
  F (suc (jump F α))                    ∎

jump-lim : jump F (lim f) ≡ lim λ n → jump F (f n)
jump-lim = refl

不动点的枚举

fixpt : (Ord → Ord) → Ord → Ord
fixpt F = jump (iterω F)

fixpt-0 : fixpt F 0 ≡ iterω F 0
fixpt-0 = refl

fixpt-suc : fixpt F (suc α) ≡ iterω F (suc (fixpt F α))
fixpt-suc {F} {α} = refl

fixpt-lim : fixpt F (lim f) ≡ lim λ n → fixpt F (f n)
fixpt-lim = refl

ε， ζ， η 层级

ε : Ord → Ord
ε = fixpt (ω ^_)

ε-0 : ε 0 ≡ iterω (ω ^_) 0
ε-0 = refl

ε-suc : ε (suc α) ≡ iterω (ω ^_) (suc (ε α))
ε-suc = refl

ε-lim : ε (lim f) ≡ lim λ n → ε (f n)
ε-lim = refl

ζ : Ord → Ord
ζ = fixpt ε

η : Ord → Ord
η = fixpt ζ
